Collection (集合函数)​

函数描述示例
array(String value1 [,String value2, ....])- 从值列表创建数组array('he',7,'xxd') = [he, 7, xxd]
array(array('he',5),'xxd') = [[he, 5], xxd]
array(array('he',5),array('','')) = [[he, 5], [, ]]
array_append(Array array, Object element)- 如果任一参数为 NULL，返回 ""
- 将元素追加到数组末尾		array_append(array('he',7,'xxd'), 'cloud') = [he, 7, xxd, cloud]
array_concat([Array array1, Array array2, ...])- 如果任一输入数组为 NULL，返回 ""
- 返回连接至少一个数组的结果		array_concat(array('he',7),array('xxd', 'cloud')) = [he, 7, xxd, cloud]
array_contains(Array array, Object needle)- 如果 'array' 为 NULL，返回 ""
- 返回给定元素是否存在于 'array' 中
注意：支持检查数组中的 null 元素		array_contains(array('he',7,'xxd'), 'cloud') = false
array_contains(array('he',-1,''),'') = true
array_distinct(Array array)- 如果 'array' 为 NULL，返回 ""
- 返回具有唯一元素的数组		array_distinct(array('he',-1,'he')) = [he, -1]
array_except(Array array1, Array array2)- 如果任一参数为 NULL，返回 ""
- 返回包含 'array1' 中不在 'array2' 中的元素的数组（去重）
- 如果排除后没有剩余元素，返回空数组		array_except(array('he',7,'xxd'),array('he')) = [7, xxd]
array_except(array('he',7,'xxd'),array('cloud')) = [he, 7, xxd]
array_intersect(Array array1, Array array2)- 如果任一参数为 NULL，返回 ""
- 返回包含 'array1' 和 'array2' 中共有元素的数组（去重）
- 如果没有共同元素，返回空数组		array_intersect(array('he',7,'xxd'),array('he')) = [he]
array_intersect(array('he',7,'xxd'),array('cloud')) = []
array_join(Array array, String delimiter[, String nullReplacement])- 如果任一参数为 NULL，返回 ""
- 使用 'delimiter' 连接数组元素，空数组元素用 'nullReplacement' 填充
注意：如果未指定 nullReplacement，数组中的 null 元素将从结果字符串中省略		array_join(array('he',7,'xxd'),'~') = he~7~xxd
array_join(array('he',3,''),'~','oo') = he~3~oo
array_max(Array array)- 如果任一参数为 NULL，返回 ""
- 返回数组中的最大值		array_max(array(4,3,56)) = 56
array_min(Array array)- 如果任一参数为 NULL，返回 ""
- 返回数组中的最小值		array_min(array(4,3,56)) = 3
array_position(Array array, Object element)- 如果任一参数为 NULL，返回 ""
- 返回 'element' 在数组中第一次出现的位置（从 1 开始）
- 如果未找到，返回 0		arrayposition(array('he',7,'xxd'),'he') = 1
array_position(array('he',7,''),'') = 0
array_prepend(Array array, Object element)- 如果 'array' 为 NULL，返回 ""
- 将元素添加到数组开头		array_prepend(array(4,3),3) = [3, 4, 3]
array_remove(Array array, Object element)- 如果 'array' 为 NULL，返回 ""
- 从数组中移除所有等于 'element' 的元素		array_remove(array('he',7,'xxd'),'he') = [7, xxd]
array_reverse(Array array)- 如果 'array' 为 NULL，返回 ""
- 返回反转顺序的数组		array_reverse(array('he',7,'xxd')) = [xxd, 7, he]
array_slice(Array array, Integer start_offset[, Integer end_offset])- 如果 'array' 或 'start_offset' 为 NULL，返回 ""
- 返回从 'start_offset' 到 'end_offset'（包含）的子数组
- 如果 'start_offset' 在 'end_offset' 之后或两者都超出数组边界，返回空数组
注意：如果省略 'end_offset'，则视为数组长度。正值从数组开头计数，负值从末尾计数		array_slice(array('he',7,'xxd'),1,2) = ['he', 7]
array_slice(array('he','xxd','b'),-2,-1) = [3, 'xxd']
array_sort(Array array[,Boolean ascending_order[,Boolean null_first]])- 如果 'array' 为 NULL，返回 ""
- 返回排序后的数组
注意：默认升序排序，NULL 在开头。'ascending_order' 为 true 时升序（NULL 在前），false 时降序（NULL 在后）。'null_first' 独立控制 NULL 位置		array_sort(array('he',7,'xxd')) = [7, he, xxd]
array_sort(array(3,7,5)) = [3, 5, 7]
array_sort(array(,3,7),false,false) = [7, 3, ]
array_sort(array(3,7,),true,false) = [3, 7, ]
array_union(Array array1[, Array array2, ...])- 如果任一输入数组为 NULL，返回 ""
- 返回连接至少一个数组的结果		array_concat(array('he',7),array('xxd', 'cloud')) = [he, 7, xxd, cloud]
cardinality(Object input)- 如果输入为 NULL，返回 ""
- 输入为数组时返回元素数量
- 输入为映射时返回条目数量		cardinality(array('he',7,'xxd')) = 3
cardinality(map('he',7,'xxd',3)) = 2
element(Array array)- 如果 'array' 为空或 NULL，返回 ""
- 返回数组的唯一元素（基数应为 1）
注意：如果数组有多个元素会抛出异常		element(array('he')) = he
find_in_set(String str, String strList)- 如果任一参数为 NULL，返回 ""
- 如果 'str' 不在 'strList' 中或 'strList' 为空字符串，返回 0
- 如果字符串 'str' 在由 N 个子字符串组成的字符串列表 'strList' 中，返回 1 到 N 范围内的值
注意：strList 是由 ',' 字符分隔的子字符串组成的字符串。如果第一个参数包含逗号字符，此函数可能无法正常工作		FIND_IN_SET('b','a,b,b,c,d') = 2
FIND_IN_SET('','a,,b,c,d') = 2
map([String value1, String value2, ...])- 如果参数数量不是偶数，返回 ""
- 从键值对列表创建映射		Map('he',7,'xxd') = null
Map('he',1,'xxd','cloud') = {he=1, xxd=cloud}
Map('xxd','cloud',map(1,2),map(3,'apple')) = {xxd=cloud, {1=2}={3=apple}}
map_entries(Map map)- 如果 'map' 为 NULL，返回 ""
- 返回给定映射中的所有条目数组		map_entries(Map('he',1,'xxd','cloud')) = [he=1, xxd=cloud]
map_entries(Map(1,2,'cloud','xxd')) = [xxd=cloud, 1=2]
map_from_arrays(Array array_of_keys, Array array_of_values)- 如果任一参数为 NULL，返回 ""
- 从键数组和值数组创建映射		map_from_arrays(array('he', 'xxd'),array(1, 3)) = {he=1, xxd=3}
map_from_arrays(array('xxd', array('cloud')),array(1, array(2))) = {1=xxd, [2]=[cloud]}
map_keys(Map map)- 如果 'map' 为 NULL，返回 ""
- 以数组形式返回映射的键（不保证顺序）		map_keys(Map('he',1,'xxd','cloud')) = [he, xxd]
map_keys(Map('xxd','cloud',map(1,2),map(3,'apple'))) = [xxd, {1=2}]
map_union([Map map1, Map map2, ...])- 如果任一映射为 NULL，返回 ""
- 通过合并至少一个映射创建新映射。这些映射应具有相同的映射类型，后面的映射会覆盖前面的映射		map_union(map('he', 1),map('xxd', 3)) = {he=1, xxd=3}
map_union(map('he', 1),map('he', 3)) = {he=3}
map_values(Map map)- 如果 'map' 为 NULL，返回 ""
- 以数组形式返回映射的值（不保证顺序）		map_values(Map('he',1,'xxd','cloud')) = [1, cloud]
map_values(Map('xxd','cloud',map(1,2),map(3,'apple'))) = [cloud, {3=apple}]

Compression (压缩函数)​

函数描述示例
compress(String string_to_compress [, String compress_type])- 如果 'string_to_compress' 为 NULL，返回 ""
- 如果 'string_to_compress' 为 ""，返回 ""
- 返回二进制字符串结果
注意：支持三种压缩算法：deflater、gzip 和 zip。'compress_type' 默认为 deflater。为了以字符串形式输出压缩结果，此方法使用 ISO_8859_1 字符集		length(compress(replicate(string1,100)),'ISO_8859_1') = 33
length(compress(''),'ISO_8859_1') = 0
uncompress(String string_to_uncompress, String compress_type)- 如果 'string_to_uncompress' 为 NULL，返回 ""
- 如果 'string_to_uncompress' 为 ""，返回 ""
- 返回字符串结果
注意：支持三种压缩算法：deflater、gzip 和 zip。'compress_type' 默认为 deflater		uncompress(compress('inlong')) = "inlong"

Condition (条件函数)​

函数描述示例
coalesce(String value1 [, String value2, ...])- 如果所有参数都为 NULL 或 ""，返回 ""
- 返回第一个不为 NULL 或 "" 的参数
注意：返回类型是其所有参数的限制最少的公共类型。如果所有参数都可为空，返回类型也可为空		coalesce('', 'SQL', 'hh') = "SQL"
if(Expr expr1, Expr expr2, Expr expr3)- 如果 'expr1' 返回 true，返回 'expr2'
- 否则返回 'expr3'		if(1 = 1,true,false) = true
ifnull(Expr expr1, Expr expr2)
if_null(Expr expr1, Expr expr2)		- 如果 'expr1' 不为 NULL，返回 'expr1'
- 否则返回 'expr2'		ifnull(null, 3) = 3
ifnull(6,'YES') = 6
isnull(Expr expr)- 如果 'expr' 为 NULL，返回 true
- 否则返回 false		isnull(5 + 3) = false
isnull(5 / 0) = true
nullif(Expr expr1, Expr expr2)
null_if(Expr expr1, Expr expr2)		- 如果 'expr1' = 'expr2' 为 true，返回 ""
- 否则返回 'expr1'		nullif(5, 3) = 5

decode(Binary binary, String charsetStr)- 如果任一参数为 NULL，返回 ""

Temporal (时间日期函数)​

函数描述示例
convert_tz(String datetime_str, String from_tz, String to_tz)- 如果任一参数为 NULL，返回 NULL
- 将日期时间字符串 'datetime_str'（默认 ISO 时间戳格式 yyyy-MM-dd HH:mm:ss）从时区 'from_tz' 转换为时区 'to_tz'
注意：时区格式应为缩写（如 "PST"）、全名（如 "America/Los_Angeles"）或自定义 ID（如 "GMT-08:00"）		CONVERT_TZ('1970-01-01 00:00:00', 'UTC', 'America/Los_Angeles') = "1969-12-31 16:00:00"
date_add(String dateStr, String intervalExprStr)- 如果 'dateStr' 为 NULL，返回 ""
- 如果 'dateStr' 是 DATE 值且计算仅涉及 YEAR, MONTH, DAY 部分（即不涉及时间部分），返回 DATE 类型
- 如果 'dateStr' 是 TIME 值且计算仅涉及 HOURS, MINUTES, SECONDS 部分（即不涉及日期部分），返回 TIME 类型
- 如果第一个参数是 DATETIME（或 TIMESTAMP）值，或者第一个参数是 DATE 但单位值使用了 HOURS, MINUTES, SECONDS，或者第一个参数是 TIME 类型但单位值使用了 YEAR, MONTH, DAY，返回 DATETIME 类型
- 否则返回 String 类型（VARCHAR）
注意：关于 intervalExpr，请参考 MySQL 官网		date_add('2020-12-31 23:59:59',INTERVAL 999 DAY) = "2023-09-26 23:59:59"
DATE_ADD('1992-12-31 23:59:59', INTERVAL '-1.999999' SECOND_MICROSECOND) = "1992-12-31 23:59:57.000001"
date_format(String timestampStr, String formatStr)- 如果任一参数为 NULL，返回 ""

注意：格式字符串与 Java 的 SimpleDateFormat 兼容		date_format('2024-08-01 22:56:56', 'yyyy/MM/dd HH:mm:ss') = "2024/08/01 22:56:56"
date_sub(String dateStr, String intervalExprStr)
datesub(String dateStr, String intervalExprStr)		- 如果 'dateStr' 为 NULL，返回 ""

- 如果日期参数是 TIME 值且计算仅涉及 HOURS, MINUTES, SECONDS 部分（即不涉及日期部分），返回 TIME 类型

- 否则返回 String 类型（VARCHAR）
注意：关于 'intervalExpr'，请参考 MySQL 官网		DATE_SUB('1992-12-31 23:59:59', INTERVAL '-1.999999' SECOND_MICROSECOND) = "1993-01-01 00:00:00.999999"
